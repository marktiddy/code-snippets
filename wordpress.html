    Custom Post Type

    You might declare this in a mu-plugins folder or in a plugin

    register_post_type('GIVE_IT_A_NAME', array(
    'show_in_rest' => true,
    'supports' => array('title', 'editor'),
    'capability_type' => 'GIVE_IT_A_NAME',
    'map_meta_cap' => true,
    'rewrite' => array(
    'slug' => 'GIVE_IT_A_NAME', ),
    'has_archive' => true,
    'public' => true, 'labels' => array(
    'name' => 'GIVE_IT_A_NAME',
    'add_new_item' => 'Add New GIVE_IT_A_NAME',
    'edit_item' => 'Edit GIVE_IT_A_NAME',
    'all_items' => 'All GIVE_IT_A_NAME',
    'singular_name' => 'GIVE_IT_A_NAME', ),
    'menu_icon' => 'dashicons-PICK-ICON',
    ));

    Enqueue a Stylesheet

    Typically goes in functions.php within a function that is called with add_action

    wp_enqueue_style('font-awesome', '//maxcdn.bootstrapcdn.com/font-awesome/4.7.0/css/font-awesome.min.css');

    Enqueue a JavaScript File

    Typically goes in functions.php within a function that is called with add_action

    wp_enqueue_script('NAME_OF_FILE', 'http://localhost:3000/bundled.js', null, '1.0', true);

    Add an Action from a Function

    add_action('wp_enqueue_scripts', 'FUNCTION_NAME');
